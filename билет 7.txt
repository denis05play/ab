using System;

class Program
{
    static void Main(string[] args)
    {
        // Создаем объекты различных фигур
        IShape circle = new Circle(5);
        IShape rectangle = new Rectangle(4, 6);
        IShape triangle = new Triangle(3, 4, 5);

        // Выводим информацию о фигурах
        Console.WriteLine("Круг:");
        Console.WriteLine($"Площадь: {circle.CalculateArea():F2}");
        Console.WriteLine($"Периметр: {circle.CalculatePerimeter():F2}\n");

        Console.WriteLine("Прямоугольник:");
        Console.WriteLine($"Площадь: {rectangle.CalculateArea():F2}");
        Console.WriteLine($"Периметр: {rectangle.CalculatePerimeter():F2}\n");

        Console.WriteLine("Треугольник:");
        Console.WriteLine($"Площадь: {triangle.CalculateArea():F2}");
        Console.WriteLine($"Периметр: {triangle.CalculatePerimeter():F2}\n");
    }
}

// Интерфейс IShape
interface IShape
{
    double CalculateArea();      // Метод для вычисления площади
    double CalculatePerimeter(); // Метод для вычисления периметра
}

// Класс Circle (круг)
class Circle : IShape
{
    public double Radius { get; set; } // Радиус круга

    // Конструктор
    public Circle(double radius)
    {
        Radius = radius;
    }

    // Реализация метода для вычисления площади
    public double CalculateArea()
    {
        return Math.PI * Radius * Radius;
    }

    // Реализация метода для вычисления периметра (длины окружности)
    public double CalculatePerimeter()
    {
        return 2 * Math.PI * Radius;
    }
}

// Класс Rectangle (прямоугольник)
class Rectangle : IShape
{
    public double Width { get; set; }  // Ширина
    public double Height { get; set; } // Высота

    // Конструктор
    public Rectangle(double width, double height)
    {
        Width = width;
        Height = height;
    }

    // Реализация метода для вычисления площади
    public double CalculateArea()
    {
        return Width * Height;
    }

    // Реализация метода для вычисления периметра
    public double CalculatePerimeter()
    {
        return 2 * (Width + Height);
    }
}

// Класс Triangle (треугольник)
class Triangle : IShape
{
    public double SideA { get; set; } // Сторона A
    public double SideB { get; set; } // Сторона B
    public double SideC { get; set; } // Сторона C

    // Конструктор
    public Triangle(double sideA, double sideB, double sideC)
    {
        SideA = sideA;
        SideB = sideB;
        SideC = sideC;
    }

    // Реализация метода для вычисления площади (по формуле Герона)
    public double CalculateArea()
    {
        double p = (SideA + SideB + SideC) / 2; // Полупериметр
        return Math.Sqrt(p * (p - SideA) * (p - SideB) * (p - SideC));
    }

    // Реализация метода для вычисления периметра
    public double CalculatePerimeter()
    {
        return SideA + SideB + SideC;
    }
}
